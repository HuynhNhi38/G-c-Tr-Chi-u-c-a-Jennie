// ╔═══════════════════════════════════════════════════════╗ // ║ Góc Trà Chiều - Google Apps Script Backend (v2) ║ // ║ Dán toàn bộ code này vào Apps Script, lưu & deploy║ // ╚═══════════════════════════════════════════════════════╝ function doGet(e) { const callback = e.parameter && e.parameter.callback; let result; try { const sheet = getSheet(); const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) { result = { stories: [] }; } else { const headers = rows[0]; const stories = rows.slice(1) .filter(r => r[0]) .map(row => { const obj = {}; headers.forEach((h, i) => { obj[h] = row[i] !== undefined ? String(row[i]) : ''; }); return obj; }); result = { stories }; } } catch (err) { result = { error: err.toString() }; } const json = JSON.stringify(result); // JSONP support (bypasses CORS) if (callback) { return ContentService.createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const sheet = getSheet(); if (body.action === 'save') { upsert(sheet, body.story); return respond({ success: true }); } if (body.action === 'delete') { del(sheet, body.id); return respond({ success: true }); } return respond({ error: 'Unknown action' }); } catch (err) { return respond({ error: err.toString() }); } } function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let s = ss.getSheetByName('stories'); if (!s) { s = ss.insertSheet('stories'); s.appendRow(['id','series','chapNum','title','category','originalAuthor','date','createdAt','releaseAt','password','content']); s.setFrozenRows(1); } return s; } function upsert(sheet, story) { const COLS = ['id','series','chapNum','title','category','originalAuthor','date','createdAt','releaseAt','password','content']; const row = COLS.map(c => story[c] !== undefined ? story[c] : ''); const lastRow = sheet.getLastRow(); if (lastRow > 1) { const ids = sheet.getRange(2, 1, lastRow - 1, 1).getValues().flat(); const idx = ids.indexOf(story.id); if (idx >= 0) { sheet.getRange(idx + 2, 1, 1, COLS.length).setValues([row]); return; } } sheet.appendRow(row); } function del(sheet, id) { const lastRow = sheet.getLastRow(); if (lastRow < 2) return; const ids = sheet.getRange(2, 1, lastRow - 1, 1).getValues().flat(); const idx = ids.indexOf(id); if (idx >= 0) sheet.deleteRow(idx + 2); } function respond(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
